vue组件的几种调用方式

2023/3/15 vue

# 1.状态管理

通过v-model或者.sync显式控制组件显示隐藏

例: 下面是element里的对话框组件的使用方法，通过 dialogVisible 状态来管理组件的显示与隐藏。

<el-button type="text" @click="dialogVisible = true">点击打开 Dialog</el-button>

<el-dialog
    title="提示"
    :visible.sync="dialogVisible"
    width="30%"
    :before-close="handleClose">
      <span>这是一段信息</span>
      <span slot="footer" class="dialog-footer">
      <el-button @click="dialogVisible = false">取 消</el-button>
      <el-button type="primary" @click="dialogVisible = false">确 定</el-button>
      </span>
</el-dialog>

<script>
  export default {
    data() {
      return {
        dialogVisible: false
      };
    },
    methods: {
      handleClose(done) {
        this.$confirm('确认关闭？')
          .then(_ => {
            done();
          })
          .catch(_ => {});
      }
    }
  };
</script>

# 2.js命令式

也就是通过js代码来调用显示组件，比如 message(消息提示，警告框)，loading等等； 主要是使用 extends(vue3废弃 => createVNode) 来创建一个组件的构造器，手动 $mount 到节点上

例：

import Vue from "vue"
import MyLoading from "../pages/index/components/myLoading.vue"

let instans = null;
const Loading = Vue.extend(MyLoading); // 生成构造器

function insertDom(box) {// 加入dom
const vm = instans.$mount();
instans.$_box = box ? box : undefined;
instans.fullscreen = box ? false : true;
instans.$_show = true
if (box) { //  box loading挂载的节点，默认body
  return document.querySelector(box).appendChild(vm.$el)
}
document.body.appendChild(vm.$el);
}

export function myLoading(option={}) { // 单例保证只有一个loading实例
const {text,box} = option;
if (instans) {
  instans.text = text
  instans.$_show && closeLoading()
  insertDom(box)
  return
}
instans = new Loading({
  data: {
    text
  }
});
insertDom(box)
}
// 关闭loading
export function closeLoading(callback, content = null) {
if (!instans) return;
const box = instans.$_box
box ? document.querySelector(box).removeChild(instans.$el) : document.body.removeChild(instans.$el)
instans.$_show = false
if (typeof callback === "function") callback.call(content)
}

export default {
install: vue => { // 挂载到根实例上
  vue.prototype.$showLoading = myLoading; // loading显示
  vue.prototype.$closeLoading = closeLoading // 关闭
}
}


//main.js 引入注册
Vue.use(loading)

// 使用
this.$showLoading({box:'.box',text:'加载中'})

vue3

import Model from "@/views/ModelView.vue"
import { ref, createVNode, render } from "vue"

let div = document.createElement("div")
document.body.appendChild(div)

export default function () {
    const isShow = ref(false)

    function show(data) {
        return new Promise((resolve, reject) => {
            function close() {
                render(null, div)
                reject()
            }
            function ok(...arg) {
                render(null, div)
                resolve(arg)
            }
            const vnode = createVNode(Model, { isShow, ok, close,data })// 组件prop里接收这些参数
            render(vnode, div)
        })
    }

    return [show]
}

# 3.指令

同样以loading为例子，比如element 的v-loading,值为true/false来控制显示隐藏，也跟上面的例子差不多，这里也利用上面的loading来继续 实现一下,把方法引入一下就可以了

insertDom 方法加一下判断，因为指令直接拿到的是节点，不是一个字符串

function insertDom(box) {
  const vm = instans.$mount();
  instans.$_box = box ? box : undefined;
  instans.fullscreen = box ? false : true;
  instans.$_show = true
  // 判断box类型
  if (Object.prototype.toString.call(box).slice(8, -1).indexOf('Element') != -1) {
    return box.appendChild(vm.$el)
  }
  
  if (box) { //  box loading挂载的节点，默认body
    return document.querySelector(box).appendChild(vm.$el)
  }
  document.body.appendChild(vm.$el);
}

注册指令调用方法即可

import { myLoading, closeLoading } from "./loading"
export default {
  install(vue) {
    vue.directive('loading', {
      bind(el, { value }) { // 绑定时
        if (value) {
          myLoading({ box: el })
        }
      },
      unbind(el, { value }) {// 解绑时
         closeLoading()
      },
      update(el, {value}) {// 更改时
        if (!value){
            closeLoading()
        }else{
            myLoading({ box: el })
        }
      }
    })
  }
}

//main.js 注册
import vLoading from "./vLoading"
Vue.use(vLoading)

// 使用
<div v-loading='true'></div>
